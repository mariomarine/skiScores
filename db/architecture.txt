# Races
- raceid integer key primary
- location varchar(30)
- date date
- class varchar(10)
- level varchar(3)

# Results
- resultid integer key primary
- racerid integer key foreign
- courseid integer key foreign
- grade integer
- team varchar(20)
- place integer
- time time


# Racers
- racerid integer key primary
- gender varchar(1)
- first varchar(20)
- last varchar(20)
- graduation integer
